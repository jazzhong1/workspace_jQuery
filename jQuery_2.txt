========================================문서객체설정========================================

--------------------------------------------------------------------each메소드---------------------------------------------------------------
jQuery배열관리

§ 배열을 관리하는 for in문과 비슷한 메소드로 객체나 배열의 요소를 검사하는 메소드

index : 객체, 배열의 순번
item : 값을 보관하는 변수(객체) ‘ . ’으로 접근

$.each(배열이름,function(index, item) { } );
지정한 배열을 0번 부터 자동으로 불러와 순번을 index, 값을 item에 넣는 메소드
ex)
$.each(배열(객체)이름, function(index, item){
	배열(객체) 차례로 처리할 때 사용
 	}	
);


$(‘s’).each(function(index,item){ } );
선택자로 선택한 요소를 index순번으로item에 요소 값을 수정시 사용하는 메소드
ex)
$(‘선택자’).each(function(index, item){
	선택된 요소들을 차례로 조작할때 사용
	}
);	
);
--------------------------------------------------------------------extend메소드---------------------------------------------------------------
jQuery에서 객체의 내용을 추가하는 메소드

$.extend(객체이름, {
	추가할 객체내용,
	추가할 객체내용2, 
	……
	}
);
--------------------------------------------------------------------noConflict메소드---------------------------------------------------------------
 jQuery변수를 $외에 다른것으로 지정하는 메소드

다른 javascript플러그인을 사용시 특수문자 $변수를 중복되어 발생하는 에러방지를 위해 사용하기도 함

var 변수명 = $.noConflict();
변수명을 $대신 사용
=========================================클래스/속성 설정==========================================

--------------------------------------------------------------------addClass()---------------------------------------------------------------
jQuery class 추가

선택자에 의해 선택된 요소에 클래스를 추가하는 메소드

1. $(‘s’).addClass(“클래스이름”);
//클래스를 여러 개 추가는 띄어쓰기로 구분

2. $(‘s’).addClass(function(n){
인자는 선택된 요소들의 번호;
return 값; // 리턴된 값을 클래스로 추가
});
--------------------------------------------------------------------removeClass()---------------------------------------------------------------
선택자에 의해 선택된 요소에 클래스를 삭제하는 메소드

1. $(‘s’).removeClass(“클래스이름”);
//클래스를 여러 개 추가는 띄어쓰기로 구분

2. $(‘s’).removeClass(function(n){
인자는 선택된 요소들의 번호;
return 값; //리턴된 값과 일치하는 클래스삭제
});
--------------------------------------------------------------------removeClass()---------------------------------------------------------------
toggleClass()
선택된 요소에 class가 있으면 삭제하고 없으면 추가하는 메소드

1. $(‘s’).toggleClass(“클래스이름”);
//클래스를 여러 개 추가는 띄어쓰기로 구분

2. $(‘s’).toggleClass(‘클래스명’, [function(index,curclass){
}],[true|false]);
☞ true : 추가만, false : 삭제만
